# music.megabyte.space — robots policy
# Last updated 2026-06-10

# Default crawlers welcome to the public catalog. Block /api/* (mostly POSTs)
# and the in-app SPA route helpers that have no SEO value of their own.
User-agent: *
Allow: /
Disallow: /api/
Disallow: /share-target
Disallow: /spotify/callback
Disallow: /embed/

# --- Search engines (full access) ---
User-agent: Googlebot
Allow: /

User-agent: Bingbot
Allow: /

User-agent: DuckDuckBot
Allow: /

# --- AI SEARCH / RETRIEVAL bots: ALLOW ---
# These power live citation in ChatGPT, Claude, Perplexity, and Google AI
# Overviews. Allowing them keeps the catalog quotable; they do NOT train models.
User-agent: OAI-SearchBot
Allow: /

User-agent: ChatGPT-User
Allow: /

User-agent: Claude-User
Allow: /

User-agent: Claude-SearchBot
Allow: /

User-agent: PerplexityBot
Allow: /

User-agent: Perplexity-User
Allow: /

User-agent: Applebot
Allow: /

User-agent: cohere-ai
Allow: /

# --- AI TRAINING bots: DISALLOW ---
# Opt out of model training (no citation benefit). This matches the
# Cloudflare-managed `ai-train=no` content signal so the two policy blocks
# agree instead of contradicting each other.
User-agent: GPTBot
Disallow: /

User-agent: ClaudeBot
Disallow: /

User-agent: Google-Extended
Disallow: /

User-agent: Applebot-Extended

User-agent: CCBot
Disallow: /
